概念

具有 一对一 逻辑关系的数据按照次序连续存储到一整块物理空间上”的存储结构就是顺序存储结构（顺序表）

#include<stdio.h>
#include<stdlib.h>
#include<time.h>
#define MAXSIZE 100
#define OK 1
#define ERROR 0
typedef int Status;
typedef int elem;


typedef struct Sqlist{
    elem *e;//存储空间的及地址
    int length;//长度
}Sqlist;

功能的实现

顺序表的初始化

//数组的初始化
Status InitArray(Sqlist *L){
    if(L->e) {//判断指针是否为空
        free(L->e);
        L->e = NULL;
    }

    L->e = (elem*)malloc(sizeof(int) * MAXSIZE);//申请capacity大小的空间

    if(L->e == NULL){
        printf("内存分配失败!\n");//判断内存分配是否成功
        exit(0);
    }

    L->length = 0;
    return OK;
}

元素的添加

//添加元素
Status PushBackArray(Sqlist *L,elem value){
    if(L->e == NULL){
        printf("请初始化顺序表\n");
        return ERROR;
    }

    if(L->length == MAXSIZE) {
        return ERROR;
    }
    L->e[L->length] = value;
    ++L->length;
    return OK;
}

元素的插入

//插入
Status InsertArray(Sqlist *L,int index,elem value){
    if(index > L->length +1 || index < 1){
        return ERROR;
    }

    for (int i = L->length; i >= index; i--) {
		L->e[i] = L->e[i - 1];
	}

	L->e[index] = value;

	++L->length;
	return OK;
}

元素的查找

//根据位置查找
elem FindByIndex(Sqlist* L,int index){
    return L->e[index];
}
//根据值查找
Status FindByValue(Sqlist* L,elem value){
    for (int i = 0; i < L->length; i++){
        if(L->e[i] == value){
            return i;
        }
    }
    return -1;
}

元素的删除

//根据值删除
Status RemoveByValue(Sqlist* L,elem value){
    for (int i = 0; i < L->length; i++){
        if(L->e[i] == value){
	        for (; i < L->length; i++) {
		        L->e[i - 1] = L->e[i];
	        }

	        --L->length;
	        return OK;
        }
    }
}
//根据位置删除
Status RemoveByIndex(Sqlist* L,int index){
	if (index > L->length + 1 || index < 1) {
		return ERROR;
	}

	for (int i = index; i < L->length; i++) {
		L->e[i - 1] = L->e[i];
	}

	--L->length;
	return OK;
}

元素的清空

//清空
Status CleanArray(Sqlist *L){
    if(L->e == NULL) {
        printf("未初始化\n");
        return ERROR;
    
    }

    L->length = 0;
    return OK;
}

线性表的销毁

//销毁
Status FreeArray(Sqlist* L){
    if(L->e != NULL){
        free(L->e);
        L->e = NULL;
    }

    L->length = 0;

    return OK;
}

顺序表的合并

//合并
Status CombineArray(Sqlist *L1,Sqlist *L2){
    if (L1->length + L2->length > MAXSIZE) {
        printf("空间不足，合并失败\n");
		return ERROR;
	}

	for (int i = 0; i < L2->length; i++) {
		L1->e[L1->length + i] = L2->e[i];
	}
    L1->length +=L2->length;

    free(L2->e);
    L2->length = 0;

    return OK;
}

顺序表的打印

//打印
void PrintArray(Sqlist* L){
    for (int i = 0; i < L->length; i++){
        printf("%d ",L->e[i]);
    }
    printf("\n");
}
void test(){//测试
    Sqlist L,L2;
      /******************************/
     /********初始化及添加元素********/
    /******************************/
    InitArray(&L);//初始化顺序表
    InitArray(&L2);
    //srand(time(NULL));
    for(int i = 0; i < 10; i++){//赋值
        PushBackArray(&L,i);
        PushBackArray(&L2,i);
    }
    printf("初始化...\n");
    PrintArray(&L);//打印
    printf("%d\n", GetLength(&L));//获取当前长度

     /******************************/
     /************插入元素***********/
    /******************************/
    InsertArray(&L,5,100);//插入元素
    printf("插入元素...\n");
    PrintArray(&L);//打印
    printf("%d\n", GetLength(&L));//获取当前长度

      /******************************/
     /************查找元素***********/
    /******************************/
    printf("查找元素...\n");
    printf("%d ",FindByIndex(&L,10));//根据位置查找
    printf("%d ",FindByValue(&L,4));//根据值查找



      /******************************/
     /************删除元素***********/
    /******************************/
    RemoveByIndex(&L,3);//根据位置删除
    RemoveByValue(&L,9);//根据值删除
    printf("删除元素...\n");
    PrintArray(&L);//打印
    printf("%d\n", GetLength(&L));//获取当前长度


      /******************************/
     /***********合并顺序表**********/
    /******************************/
    CombineArray(&L,&L2);//合并
    printf("合并顺序表...\n");
    PrintArray(&L);//打印
    printf("%d\n", GetLength(&L));//获取当前长度

     /******************************/
     /***********清空顺序表**********/
    /******************************/
    CleanArray(&L);//清空
    printf("清空顺序表...\n");
    PrintArray(&L);//打印
    printf("%d\n", GetLength(&L));//获取当前长度

      /******************************/
     /***********销毁顺序表**********/
    /******************************/
    FreeArray(&L);//销毁
    printf("销毁顺序表...\n");
    printf("%d",PushBackArray(&L,1));
}


int main(){
    test();
    system("pause");
    return 0;
}
